async function uploadToGitHub() { if (!processedImageData) return alert("Tire uma foto primeiro."); const base64 = processedImageData.split(",")[1]; const fileName = "foto_" + Date.now() + ".png"; const url = `https://api.github.com/repos/${GH_USER}/${GH_REPO}/contents/${GH_PATH}/${fileName}`; const payload = { message: "Upload autom√°tico do app Tons", content: base64, branch: GH_BRANCH }; const response = await fetch(url, { method: "PUT", headers: { "Authorization": `Bearer ${GH_TOKEN}`, "Content-Type": "application/json" }, body: JSON.stringify(payload) }); if (response.ok) { alert("Imagem enviada com sucesso ao GitHub!"); } else { const error = await response.json(); console.error(error); alert("Erro ao enviar: " + error.message); } }
